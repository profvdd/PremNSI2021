def top_prenom2(liste1,annee,eff_min):
    dico2={}
    for ligne in liste1:
        if int(ligne['annais'])==annee and int(ligne['nombre'])>eff_min:
            sexe=ligne['sexe']
            prenom=ligne['preusuel']
            eff=int(ligne['nombre'])
            dico2[(sexe,prenom)]=eff
    return dico2
